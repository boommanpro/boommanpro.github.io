  1. jps 进程目录获取
#!/bin/bash

all=`jps |grep -v 'Jps'|awk '{print $1}'`

for o in $all
do
  echo $o "->" `ls -l /proc/$o |grep 'cwd' |awk '{print $11}'`
done

    rpm -e --nodeps `rpm -qa | grep -i openJDK`
fi
